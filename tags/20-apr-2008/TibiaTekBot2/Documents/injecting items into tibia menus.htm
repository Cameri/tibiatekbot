TibiaMenu.h
Code:
#ifndef __TIBIA_MENU_H__
#define __TIBIA_MENU_H__

#include "includes.h"

// void* addr = &myFunc;
// asm("CALL <addr>");

class TibiaMenu
{
      public:
             TibiaMenu(int, DWORD);
             ~TibiaMenu();
             
             // Current version of Tibia.
             int Version;
             
             // Index of the address with our version.
             int VersionIndex;
             
             // Process Handle of the current Tibia version.
             DWORD processHandle;
             
             // Function that should be called externally (from tibia)
             // when a menu item is pressed. 
             void MenuItemPressed(int);
             
             // Function that should b called externally (from tibia)
             // when we should add menu items.
             void MenuItemsAdd();
             
             // Function to be called in our constructor to
             // initialize our code.
             void InitializeASM();
             
             // Function to write a uint on a char buffer.
             // just makes it easier for pointers and such.
             void WriteUInt(char*, int, int);
             void WriteUIntEx(char*, int, int);
};

#endif

TibiaMenu.cpp
Code:
#include "includes.h"

TibiaMenu* mTibiaMenu;

struct addr
{
       unsigned short version; // Tibia Version
       char* aCodeCave; // Suitable CodeCave location.
       char* aItemCall; // Location of the item-menu call we will replace.
       char* aItemCallNext; // Location of the next opcode following the above call.
       char* fAddMenuItem; // Location of the AddMenu function.
       char* mEventErrorLoc; // Location of the "unknown event" junk.
       char* menuListFuncEnd; // Location of the start of the POP's at the end of the menu switch case.
       char* JAUnknownEvents; // Location of the JA which would make us debug with a 'unknown events'.
};

addr mAddresses[] =
{
         { 772,
             (char*)0x004002C3, // CodeCave
             (char*)0x00443427, // Item-Menu Call (Moveable)
             (char*)0x0044342C, // Location of next command after Item-Menu Call (Moveable)
             (char*)0x00442910, // AddMenu Function Location.
             (char*)0x00421208, // "Unknown Event" function
             (char*)0x0043F413, // POP after menu-handle code.
             (char*)0x0043F3D9  // JA where "unknown event" is checked.
         },
         
         { 781,
             (char*)0x004002C3, // CodeCave
             (char*)0x00443427, // Item-Menu Call (Moveable)
             (char*)0x0044342C, // Location of next command after Item-Menu Call (Moveable)
             (char*)0x00442910, // AddMenu Function Location.
             (char*)0x00421208, // "Unknown Event" function
             (char*)0x0043F413, // POP after menu-handle code.
             (char*)0x0043F3D9  // JA where "unknown event" is checked.
         },
         
     { 0, 0, 0, 0, 0, 0, 0 }
};

char* txt = "Set Hotkey";

TibiaMenu::TibiaMenu(int v, DWORD ph)
{
     // TibiaMenu Constructor
     
     // Make ourself the "main" tibia menu.
     mTibiaMenu = this;
     
     // Set the internal version.
     Version = v;
     
     // Set the internal processHandle.
     processHandle = ph;
     
     // Get the count of the items in the Addresses.
     unsigned short count = sizeof(mAddresses)/sizeof(addr);
     
     // Attempt to find our address index.
     for (int i = 0; i < count; i++)
     {
         if (mAddresses[i].version == Version)
         {
            TibiaMenu::VersionIndex = i;                   
         }
     }    
     
     // Now initialize our ASM code.
     InitializeASM();    
}

TibiaMenu::~TibiaMenu()
{
     // TibiaMenu Deconstructor                       
}

static int addr = 0, fAddr = 0;
static int id = 0;

void TibiaMenu::MenuItemsAdd()
{
     // TODO
     addr = (int)(txt);
     fAddr = (int)mAddresses[VersionIndex].fAddMenuItem;
     id = 0x0044D6F0;     
     
         asm("PUSH ECX");
         asm("PUSH 0");
         //asm("PUSH 0x55FB24");
         asm("PUSH DWORD PTR [_addr]");
         asm("PUSH 0x9999");
         asm("MOV ECX,ESI");
         asm("CALL DWORD PTR [_fAddr]");
         asm("POP ECX");
     
     
     return;
}

static void sMenuItemsAdd()
{
     if (mTibiaMenu != NULL)
        mTibiaMenu->MenuItemsAdd();
        
     return; 
}

static void sMenuItemPressed(int index)
{
     if (mTibiaMenu != NULL)
        mTibiaMenu->MenuItemPressed(index);     
}

void TibiaMenu::WriteUInt(char* buff, int start, int x)
{
   buff[start] =   (char)( x         & 0x000000ff );
   buff[start+1] = (char)( x >> 8    & 0x000000ff );
   buff[start+2] = (char)( x >> 16   & 0x000000ff );
   buff[start+3] = (char)( x >> 24   & 0x000000ff );
}

TibiaMenu.cpp
Code:
void TibiaMenu::InitializeASM()
{
     // Here we will write the ASM, and other things, then write
     // it into memory.
     
     // First thing, get a pointer to the static sMenuItemsAdd function.
     // We'll have Tibia call it, and it will call the last initialized
     // Menu classes function.
     int fAddr = (int)(&sMenuItemsAdd);
     
     // Now we should get a pointer to the MenuItemPressed function,
     // so we can handle that.
     int gAddr = (int)(&sMenuItemPressed);
     
     // We will store a position here, a relative & displacement
     // position for jumps and calls.
     int RelPosDis = 0;
     
     // Our char buffer containing the assembly code we will use to
     // replace the CALL function at the 'aItemCall' location.
     char AsmJumpInject[5] =
     {
          0xE9, 0x00, 0x00, 0x00, 0x00 // JMP 0 ; Replace with 'aCodeCave'.
     };
     
     // However, it is not complete. We need to fill it in with
     // the address of our codecave.
     // First prepare the relative-displacement junk.
     RelPosDis = (((int)mAddresses[VersionIndex].aCodeCave) - ((int)mAddresses[VersionIndex].aItemCall)) - 5;
     WriteUInt(AsmJumpInject, 1,  RelPosDis);
     
     // The above one is calculated differently than the others
     // because it it goes up from the position, not down.
     
     // Now lets prepare our ASM code that we will inject.
     // This needs to be written to the codecave location,
     // and we need to make sure the codecave has enough space
     // for this call. We should do that before we write this though.
     char AsmInjectCode[45] =
     {
          0xE8, 0x00, 0x00, 0x00, 0x00,         // CALL 0    ; Replace with 'fAddMenuItem'.
          0xE8, 0x00, 0x00, 0x00, 0x00,         // CALL 0    ; Replace with 'fAddr'.
          0xE9, 0x00, 0x00, 0x00, 0x00,         // JMP  0    ; Replace with 'aItemCallNext'.
          
          0x90, 0x90,                           // 0000      ; Just a blank space for us.
     };
     
     /*
     char AsmInjectCode[45] =
     {
          0xE8, 0x00, 0x00, 0x00, 0x00,         // CALL 0    ; Replace with 'fAddMenuItem'.
          0xE8, 0x00, 0x00, 0x00, 0x00,         // CALL 0    ; Replace with 'fAddr'.
          0xE9, 0x00, 0x00, 0x00, 0x00,         // JMP  0    ; Replace with 'aItemCallNext'.
          
          0x90, 0x90,                           // 0000      ; Just a blank space for us.
          
          // 0x83, 0xEC, 0x04,                     // SUB ESP,4 ; Subtract 4 from ESP for our return.
          0x90, 0x90, 0x90,
          0xE8, 0x00, 0x00, 0x00, 0x00,         // CALL 0    ; Replace with 'gAddr'. Call our function.
          //0x80, 0x3C, 0x24, 0x00,               // CMP ESP,0 ; Compare ESP with 0. 0 means no errors in our call.
          0x90, 0x90, 0x90, 0x90,
          //0x83, 0xC4, 0x04,                     // ADD ESP,4 ; Add back the 4 to ESP we subtracted.
          0x90, 0x90, 0x90,
          //0x0F, 0x85, 0x00, 0x00, 0x00, 0x00,   // JNZ 0     ; Jump to this location if ESP is not 0. (This is the unknown error thing), so replace it.
          0x90, 0x90, 0x90, 0x90, 0x90, 0x90,
          0xE9, 0x00, 0x00, 0x00, 0x00,         // JMP 0     ; Replace with the function end. We'll do this if ESP is 0 (no errors).
          //0x90, 0x90                            // 0000      ; Just a blank space for us.
          0x90, 0x90
     };
     */
     
     // Now we need to do some replacements in that code to make sure
     // that everything is pointing to the right place.
     
     // First calculate the displacement and such.
     RelPosDis = (((int)mAddresses[VersionIndex].aCodeCave + 20) - gAddr) + 5;
     WriteUInt(AsmInjectCode, 21, abs(RelPosDis));
     
     // Lets start with writing the menu item stuff.
     // First calculate the displacement and such.
     RelPosDis = (((int)mAddresses[VersionIndex].aCodeCave) - (int)mAddresses[VersionIndex].fAddMenuItem) + 5;
     WriteUInt(AsmInjectCode, 1, abs(RelPosDis));
     
     // Now lets write the 'fAddr', where our own function is.
     // First calculate the displacement and such..
     RelPosDis = (((int)mAddresses[VersionIndex].aCodeCave + 5) - fAddr) + 5;
     WriteUInt(AsmInjectCode, 6, abs(RelPosDis));
     
     // Now we should write 'aItemCallNext' so after it calls our function, it returns
     // back to the original code.
     RelPosDis = (((int)mAddresses[VersionIndex].aCodeCave + 10) - (int)mAddresses[VersionIndex].aItemCallNext) + 5;
     WriteUInt(AsmInjectCode, 11, abs(RelPosDis));
     
     // Now we should write the location of the unknown error junk, so
     // if theres something wrong with Tibia, not us, the player knows it.
     RelPosDis = (((int)mAddresses[VersionIndex].aCodeCave + 33) - (int)mAddresses[VersionIndex].mEventErrorLoc) + 5;
     WriteUInt(AsmInjectCode, 34, abs(RelPosDis));
     
     // And finally (atleast for now.. maybe not in the future) we will write the location
     // of the "end of the function", where the menu handling function ends, and is cleaning up.
     RelPosDis = (((int)mAddresses[VersionIndex].aCodeCave + 38) - (int)mAddresses[VersionIndex].menuListFuncEnd) + 5;
     WriteUInt(AsmInjectCode, 39, abs(RelPosDis));
     
     // Now that our injection code is finished, we can unprotect the memory of
     // our ASM code and write it.
     
     // The old protection settings of a piece of memory.
     // We'll use this to reset the last protection setting.
     DWORD old;
     
     // Now lets allow ourselfs to write the code we will inject, from the 'aCodeCave'
     // position to the size of the AsmInjectCode array.
     VirtualProtectEx((void*)processHandle, ((char*)mAddresses[VersionIndex].aCodeCave), sizeof(AsmInjectCode), PAGE_EXECUTE_READWRITE, &old);
     
     // Now we should write our inject-code into the memory space now, in the
     // location of our code cave.
     WriteProcessMemory((void*)processHandle, (void*)mAddresses[VersionIndex].aCodeCave, &AsmInjectCode, sizeof(AsmInjectCode), NULL);
     
     // Now that we are done, we should set the protection of the memory
     // back to its old state.
     VirtualProtectEx((void*)processHandle, ((char*)mAddresses[VersionIndex].aCodeCave), sizeof(AsmInjectCode), old, &old);
     
     // Moving on, we should now unprotect the area for the JumpInject that leads
     // Tibia to our own code.
     VirtualProtectEx((void*)processHandle, ((char*)mAddresses[VersionIndex].aItemCall), sizeof(AsmJumpInject), PAGE_EXECUTE_READWRITE, &old);
     
     // Now write the code in the position of the item call ASM.
     WriteProcessMemory((void*)processHandle, (void*)mAddresses[VersionIndex].aItemCall, &AsmJumpInject, sizeof(AsmJumpInject), NULL);
     
     // Now restore the protection of that memory with the old protection.
     VirtualProtectEx((void*)processHandle, ((char*)mAddresses[VersionIndex].aItemCall), sizeof(AsmJumpInject), old, &old);
     
     // Prepare our JA modification.
     // 16 spaces offset.
     char AsmJumpAboveInject[6] =
     {
          0x0F, 0x87, 0x00, 0x00, 0x00, 0x00 // JA 0 ; Replace with the code location.
     };
     
     // Now we should calculate the relative distance and such.
     RelPosDis = (((int)mAddresses[VersionIndex].aCodeCave + 17) - ((int)mAddresses[VersionIndex].JAUnknownEvents)) - 6;
     WriteUInt(AsmJumpAboveInject, 2,  RelPosDis);
     
     // Now we should unprotect the memory, and then write 
     // the buffer.
     VirtualProtectEx((void*)processHandle, ((char*)mAddresses[VersionIndex].JAUnknownEvents), sizeof(AsmJumpAboveInject), PAGE_EXECUTE_READWRITE, &old);
     
     // Write the data now..
     WriteProcessMemory((void*)processHandle, (void*)mAddresses[VersionIndex].JAUnknownEvents, &AsmJumpAboveInject, sizeof(AsmJumpAboveInject), NULL);
     
     // Now replace the old protection settings.
     VirtualProtectEx((void*)processHandle, ((char*)mAddresses[VersionIndex].JAUnknownEvents), sizeof(AsmJumpAboveInject), old, &old);
     
     // And wallaah. Our Menu hack is completed :).
}
If you have any questions, feel free to ask.. I probably won't get the new addresses for you. The idea is simple, really... The class is designed to take some memory addresses stored in a struct, and input them into some char arrays which represent compiled x86 assembly code. The idea is that it will fill in the proper values and then write the char array into Tibia's memory, thus overwriting (or patching) the code.
